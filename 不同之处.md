# 指针空值类型

```c++
#include <iostream>
using namespace std;

void func(char *p)
{
    cout << "void func(char *p)" << endl;
}

void func(int p)
{
    cout << "void func(int p)" << endl;
}

int main()
{
    func(NULL);   // 想要调用重载函数 void func(char *p)   //这里报错  说是ambiguous
    func(250);    // 想要调用重载函数 void func(int p)

    return 0;
}
```
